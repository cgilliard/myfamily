66 // Initialize a chain_guard with the specified SlabAllocator and sync status.
67 // When the chain guard goes out of scope the the 'link' for this SlabAllocator is removed from
68 // the chain leaving the allocation chain in the state it was at before calling this function.
69 // Important: This function must not be called more than once per scope or behaviour is undefined.
70 // Example (Ok):
71 // {
72 // ChainGuard c1 = set_slab_allocator(&myallocator1, false);
73 // {
74 // ChainGuard c2 = set_slab_allocator(&myallocator2, true);
75 // // ...
76 // }
77 // }
78 // Example (NOT Ok):
79 // {
80 // ChainGuard c1 = set_slab_allocator(&myallocator1, false);
81 // ChainGuard c2 = set_slab_allocator(&myallocator2, true);
82 // // ...
83 // }
