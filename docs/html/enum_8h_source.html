31 // The enum macro builds the enum and the type string for the Enum.
32 #define Enum(name, ...) \
33  typedef enum _Enum_##name{ \
34  FOR_EACH(PROC_ENUM_DEFN, name, (), __VA_ARGS__) \
35  _Enum_##name##_size} _Enum_##name; \
36  FOR_EACH(PROC_ENUM_TYPES, name, (), __VA_ARGS__) \
37  Builder(name, Config(i32, variant_id), Config(const Obj*, value)); \
38  Impl(name, AsRef); \
39  Impl(name, EnumProps); \
40  Impl(name, Build);
41 #define EnumImpl(name) \
42  Type(name, Field(i32, variant_id), Object(Rc, value)); \
43  void name##_build(const name##Config* config) \
44  { \
45  $ContextVar($Var(), name, variant_id) = config->variant_id; \
46  let rc = new (Rc, With(value, config->value)); \
47  Move(&$ContextVar($Var(), name, value), &rc); \
48  } \
49  i32 name##_variant_id() { return $Context($(), name, variant_id); } \
50  Obj name##_as_ref() \
51  { \
52  var k = klone(&$Context($(), name, value)); \
53  let obj = unwrap(&k); \
54  ReturnObj(obj); \
55  }

145 static bool is_prim_match(const char* expected, const char* found)
146 {
147  if (!strcmp(found, "I8") && !strcmp(expected, "int8_t"))
148  return true;
149  if (!strcmp(found, "I16") && !strcmp(expected, "int16_t"))
150  return true;
151  if (!strcmp(found, "I32") && !strcmp(expected, "int32_t"))
152  return true;
153  if (!strcmp(found, "I64") && !strcmp(expected, "int64_t"))
154  return true;
155  if (!strcmp(found, "I128") && !strcmp(expected, "int128_t"))
156  return true;
157  if (!strcmp(found, "U8") && !strcmp(expected, "uint8_t"))
158  return true;
159  if (!strcmp(found, "U16") && !strcmp(expected, "uint16_t"))
160  return true;
161  if (!strcmp(found, "U32") && !strcmp(expected, "uint32_t"))
162  return true;
163  if (!strcmp(found, "U64") && !strcmp(expected, "uint64_t"))
164  return true;
165  if (!strcmp(found, "U128") && !strcmp(expected, "uint128_t"))
166  return true;
167  if (!strcmp(found, "F32") && !strcmp(expected, "float"))
168  return true;
169  if (!strcmp(found, "F64") && !strcmp(expected, "double"))
170  return true;
171  if (!strcmp(found, "Bool") && !strcmp(expected, "_Bool"))
172  return true;
173  return false;
174 }
