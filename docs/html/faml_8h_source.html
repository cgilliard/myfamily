68 // Implementation notes:
69 // These functions are for definitions which have length and perhaps default
70 // MyType {
71 // g = [u8 | 20]; // statically sized array size 20 defaults to 0
72 // h = [u8]; // dynamically sized array must be allocated using $Alloc
73 // i = [u8 | 20 | 10] // statically sized array size 20 with defaults to 10
74 // }
75 // multi dimensional array syntax: x = [[u8 | 20] | 10] // this is a statically sized array of 10
