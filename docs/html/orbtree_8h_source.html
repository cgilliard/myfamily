22 typedef struct OrbTreeNode {
23  // ordered tree
24  // upper bit indicates if this is a 'box' or 'weak' type and as such data is
25  // treated as a pointer.
26  unsigned int ord_parent;
27  // if upper bit set and upper bit of parent is set, this indicates 'weak'.
28  // if upper bit not set and upper bit of parent is set, this indicates
29  // 'box'.
30  unsigned int ord_right;
31  unsigned int ord_left;
32  unsigned int ord_right_subtree_size;
33  // upper bit for left subtree size is ord_color
34  unsigned int ord_left_subtree_size;
35 
36  // sequenced tree
37  unsigned int seq_parent;
38  unsigned int seq_right;
39  unsigned int seq_left;
40  unsigned int seq_right_subtree_size;
41  // upper bit for left subtree size is seq_color
42  unsigned int seq_left_subtree_size;
43 
44  // namespace
45  unsigned int namespace;
46  unsigned int seqno;
47  // eventually move to our string type for this (both are pointers so 8
48  // bytes)
49  char *name;
50 
51  // if this is a 'box' or 'weak' type as indicated in the upper bit of
52  // ord_parent/ord_right, we treat data as pointer. otherwise, the first byte
53  // is type indicator (int, byte, bool, float, or weak). Since all types only
54  // require 4 bytes or less the remaining 7 bytes. We use the same 'aux'
55  // reference counting for types as in the previous object model.
56  byte data[8];
57 } OrbTreeNode;
