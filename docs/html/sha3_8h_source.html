20 // Added macro to hash a string
21 #define SHA3_HASH(in, out) \
22  ({ \
23  sha3_context _c__; \
24  const byte *_hash__; \
25  sha3_Init256(&_c__); \
26  sha3_Update(&_c__, in, mystrlen(in)); \
27  _hash__ = sha3_Finalize(&_c__); \
28  for (int i = 0; i < 32; i++) { \
29  char s[23]; \
30  byte_to_hex(_hash__[i], s); \
31  out[i * 2] = s[0]; \
32  out[(i * 2) + 1] = s[1]; \
33  } \
34  out[64] = 0; \
35  })

37 static void byte_to_hex(byte b, char s[23]) {
38  unsigned i = 1;
39  s[0] = s[1] = '0';
40  s[2] = '\0';
41  while (b) {
42  unsigned t = b & 0x0f;
43  if (t < 10) {
44  s[i] = '0' + t;
45  } else {
46  s[i] = 'a' + t - 10;
47  }
48  i--;
49  b >>= 4;
50  }
51 }
