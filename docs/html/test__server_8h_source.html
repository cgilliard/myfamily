32 // In the header file, we define two traits related to our server. The first is the core functionalities of the server.
33 // This is a contrived example, but demonstrates the usage of the library fairly well.
34 // The 'DefineTrait' macro is used to define a trait. This one is called 'HttpServerCore'. It has one required method
35 // and one default implementation. Any default implementation can be overriden using the 'Override' macro.
36 #define HttpServerCore DefineTrait( \
37  HttpServerCore, \
38  Required(Const, void, print_stats), \
39  RequiredWithDefault(format_print_stats_impl, Const, void, format_print_stats))

41 // The 'TraitImpl' mcro generates headers and implementations of the calling functions.
42 // In this case 'print_stats' and 'format_print_stats'.
43 TraitImpl(HttpServerCore);

45 // Here we define a second trait. This one names 'HttpServerCore' as a super trait. Any Type implementing
46 // this trait must also implement 'HttpServerCore' as well. There are three other required methods in this trait.
47 #define HttpServerApi DefineTrait( \
48  HttpServerApi, \
49  Super(HttpServerCore), \
50  Required(Var, bool, start_server), \
