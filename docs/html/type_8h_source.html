277 #define Move(dst, src) \
278  ({ \
279  if (((*((Obj *)src)).flags & OBJECT_FLAGS_CONSUMED) != 0) \
280  panic("src object has already been consumed\n"); \
281  if (((*((Obj *)dst)).vtable != NULL && \
282  ((*((Obj *)dst)).vtable != ((*((Obj *)src)).vtable)))) { \
283  /* This is a generic */ \
284  if ((*((Obj *)dst)).vtable->is_trait_specifier) { \
285  /* Check trait bounds */ \
286  for (u64 i = 0; i < (*((Obj *)dst)).vtable->trait_len; i++) { \
287  if (!Implements((*((Obj *)src)), \
288  (*((Obj *)dst)).vtable->trait_entries[i].trait_name)) \
289  panic("required trait [%s] not implemented in type [%s].", \
290  (*((Obj *)dst)).vtable->trait_entries[i], \
291  (*((Obj *)src)).vtable->name); \
292  } \
293  } /* This is a concrete type */ \
294  else \
295  panic("Vtable mismatch! Trying to set type %s to " \
296  "type %s!", \
297  ((*((Obj *)dst)).vtable)->name, ((*((Obj *)src)).vtable)->name); \
298  } \
299  if (((*((Obj *)dst)).flags & OBJECT_FLAGS_CONSUMED) == 0) \
300  Obj_cleanup(dst); \
301  /* We copy the flags and vtable from src and the Obj data (ptr).*/ \
302  (*((Obj *)dst)).vtable = (*((Obj *)src)).vtable; \
303  (*((Obj *)dst)).flags = (*((Obj *)src)).flags; \
304  (*((Obj *)dst)).ptr = (*((Obj *)src)).ptr; \
305  /* Set src's flags to consumed and no_cleanup as the ownership is moved. */ \
306  (*((Obj *)src)).flags |= OBJECT_FLAGS_CONSUMED | OBJECT_FLAGS_NO_CLEANUP; \
307  })
