186 #define Move(dst, src) \
187  ({ \
188  if (((*((Object*)src)).flags & OBJECT_FLAGS_CONSUMED) != 0) \
189  panic("src object has already been consumed\n"); \
190  /* Check for vtable mismatch */ \
191  if (((*((Object*)dst)).vtable != NULL && ((*((Object*)dst)).vtable != ((*((Object*)src)).vtable)))) \
192  { \
193  panic("Vtable mismatch! Trying to set type %s to " \
194  "type %s!", \
195  ((*((Object*)dst)).vtable)->name, \
196  ((*((Object*)src)).vtable)->name); \
197  } \
198  /* If the object is not consumed we call cleanup as we are \
199  * overwriting it.*/ \
200  if (((*((Object*)dst)).flags & OBJECT_FLAGS_CONSUMED) == 0) \
201  Object_cleanup(dst); \
202  *((Object*)dst) = *((Object*)src); \
203  (*((Object*)src)).flags |= \
204  OBJECT_FLAGS_CONSUMED | OBJECT_FLAGS_NO_CLEANUP; \
205  })
